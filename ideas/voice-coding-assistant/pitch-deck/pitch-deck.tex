\begin{frame}{Appendix: Technical Architecture}
\small{
\textbf{Tech Stack:}
\begin{itemize}
\item Mobile: React Native + Expo SDK 52
\item STT: OpenAI Whisper API (95-98\% accuracy)
\item TTS: expo-speech (on-device, <50ms latency)
\item Backend: Fastify 4 + PostgreSQL 16
\item Real-time: Polling + Push Notifications
\item Auth: OAuth 2.0 PKCE + GitHub
\end{itemize}

\textbf{Changed from Original Plan:}
\begin{itemize}
\item ✗ expo-speech-recognition → ✓ Whisper API (quality requirement)
\item ✗ Pusher → ✓ Supabase/Polling (cost optimization)
\item ✗ Regex parser → ✓ Claude Haiku (95\%+ accuracy)
\end{itemize}
}
\end{frame}
