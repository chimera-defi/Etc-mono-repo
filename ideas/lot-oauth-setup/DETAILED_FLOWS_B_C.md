# LOT: Detailed User Flows for Options B & C

## Overview

This document provides in-depth exploration of:
- **Option B**: External Browser + Local Callback (OAuth flow)
- **Option C**: Manual API Key Entry (Fallback)

Both approaches prioritize security and work together as primary + fallback.

---

# Option B: External Browser + Local Callback

## Why This Approach

1. **Full browser security** - Password managers, hardware keys, biometrics all work
2. **Familiar OAuth pattern** - Users have done this with "Sign in with Google" countless times
3. **No webview limitations** - Some providers block embedded webviews
4. **2FA friendly** - Authenticator apps, SMS, hardware keys all supported

### Local Callback Server

```typescript
// lot-mcp-server/src/oauth/callback-server.ts

import http from 'http';
import { URL } from 'url';
import crypto from 'crypto';

interface CallbackResult {
  code: string;
  state: string;
}

interface PendingAuth {
  state: string;
  codeVerifier: string;  // For PKCE
  resolve: (result: CallbackResult) => void;
  reject: (error: Error) => void;
  timeout: NodeJS.Timeout;
}

const pendingAuths = new Map<string, PendingAuth>();

export async function startOAuthFlow(
  provider: string,
  config: OAuthConfig
): Promise<{ authUrl: string; waitForCallback: () => Promise<CallbackResult> }> {
  
  // Generate PKCE values
  const codeVerifier = crypto.randomBytes(32).toString('base64url');
  const codeChallenge = crypto
    .createHash('sha256')
    .update(codeVerifier)
    .digest('base64url');
  
  // Generate state for CSRF protection
  const state = crypto.randomBytes(16).toString('hex');
  
  // Find available port
  const port = await findAvailablePort(9876, 9900);
  const redirectUri = `http://localhost:${port}/callback`;
  
  // Build authorization URL
  const authUrl = new URL(config.authorizationEndpoint);
  authUrl.searchParams.set('client_id', config.clientId);
  authUrl.searchParams.set('redirect_uri', redirectUri);
  authUrl.searchParams.set('response_type', 'code');
  authUrl.searchParams.set('scope', config.scopes.join(' '));
  authUrl.searchParams.set('state', state);
  authUrl.searchParams.set('code_challenge', codeChallenge);
  authUrl.searchParams.set('code_challenge_method', 'S256');
  
  // Start callback server
  const server = await startCallbackServer(port);
  
  // Create promise that resolves when callback is received
  const waitForCallback = () => new Promise<CallbackResult>((resolve, reject) => {
    const timeout = setTimeout(() => {
      pendingAuths.delete(state);
      server.close();
      reject(new Error('Authentication timed out after 5 minutes'));
    }, 5 * 60 * 1000);
    
    pendingAuths.set(state, {
      state,
      codeVerifier,
      resolve: (result) => {
        clearTimeout(timeout);
        server.close();
        resolve(result);
      },
      reject: (error) => {
        clearTimeout(timeout);
        server.close();
        reject(error);
      },
      timeout
    });
  });
  
  return { authUrl: authUrl.toString(), waitForCallback };
}

function startCallbackServer(port: number): Promise<http.Server> {
  return new Promise((resolve, reject) => {
    const server = http.createServer((req, res) => {
      const url = new URL(req.url!, `http://localhost:${port}`);
      
      if (url.pathname === '/callback') {
        handleCallback(url, res);
      } else {
        res.writeHead(404);
        res.end('Not found');
      }
    });
    
    server.listen(port, 'localhost', () => {
      console.log(`OAuth callback server listening on port ${port}`);
      resolve(server);
    });
    
    server.on('error', reject);
  });
}

function handleCallback(url: URL, res: http.ServerResponse) {
  const code = url.searchParams.get('code');
  const state = url.searchParams.get('state');
  const error = url.searchParams.get('error');
  
  // Check for OAuth error
  if (error) {
    const errorDescription = url.searchParams.get('error_description') || error;
    serveErrorPage(res, errorDescription);
    
    const pending = state ? pendingAuths.get(state) : null;
    if (pending) {
      pending.reject(new Error(errorDescription));
      pendingAuths.delete(state!);
    }
    return;
  }
  
  // Validate state
  if (!state || !pendingAuths.has(state)) {
    serveErrorPage(res, 'Invalid or expired authentication session');
    return;
  }
  
  // Validate code
  if (!code) {
    serveErrorPage(res, 'No authorization code received');
    pendingAuths.get(state)?.reject(new Error('No authorization code'));
    pendingAuths.delete(state);
    return;
  }
  
  // Success!
  serveSuccessPage(res);
  
  const pending = pendingAuths.get(state)!;
  pending.resolve({ code, state });
  pendingAuths.delete(state);
}

async function findAvailablePort(start: number, end: number): Promise<number> {
  for (let port = start; port <= end; port++) {
    if (await isPortAvailable(port)) {
      return port;
    }
  }
  throw new Error(`No available ports in range ${start}-${end}`);
}

function isPortAvailable(port: number): Promise<boolean> {
  return new Promise((resolve) => {
    const server = http.createServer();
    server.listen(port, 'localhost', () => {
      server.close(() => resolve(true));
    });
    server.on('error', () => resolve(false));
  });
}

function escapeHtml(text: string): string {
  return text
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
    .replace(/'/g, '&#039;');
}
```

```typescript
// Secure key handling in MCP server
async function handleManualKeyEntry(
  provider: string,
  rawKey: string,
  conversationId: string
): Promise<ManualKeyResult> {
  const config = manualSetupProviders[provider];
  
  // 1. Validate format immediately
  if (!config.keyValidation.test(rawKey)) {
    return {
      success: false,
      error: `Invalid key format. Expected: ${config.keyFormat}`,
      // Return error without echoing the key
    };
  }
  
  // 2. Store securely IMMEDIATELY (before any other processing)
  await storeInKeychain(provider, rawKey);
  
  // 3. Test the key (optional, some users may want to skip)
  let keyInfo = null;
  try {
    keyInfo = await testKeyWithProvider(provider, rawKey);
  } catch (error) {
    // Key might be invalid, but we still stored it
    // Let user know but don't fail completely
  }
  
  // 4. Return result with MASKED key only
  return {
    success: true,
    maskedKey: maskApiKey(rawKey, provider),
    keyInfo, // Account info, permissions, etc.
    // NEVER include rawKey in response
  };
}

interface ConversationMessage {
  role: 'user' | 'assistant';
  content: string;
  metadata?: {
    containsSensitiveData?: boolean;
    excludeFromContext?: boolean;
  };
}
```

## Summary

### Option B: Browser + Local Callback

| Aspect | Details |
|--------|---------|
| **Best for** | Providers with OAuth support, security-conscious users |
| **UX** | Familiar OAuth flow, full browser security |
| **Security** | â­â­â­â­â­ - PKCE, browser security, local callback |
| **Implementation** | Local HTTP server on random port |
| **Edge cases** | Port conflicts, timeouts, user denial |

### Option C: Manual Entry

| Aspect | Details |
|--------|---------|
| **Best for** | All providers, offline setup, fallback |
| **UX** | Guided step-by-step, works anywhere |
| **Security** | â­â­â­â­ - Key masking, secure storage, no logging |
| **Implementation** | Key validation, secure storage, provider testing |
| **Edge cases** | Invalid keys, wrong permissions, format errors |

### Recommendation

**Use both together:**
1. Offer OAuth (Option B) as primary for supported providers
2. Always show manual option (Option C) as visible fallback
3. Gracefully transition to manual if OAuth fails
4. Use same file generation logic for both paths
